function [t, r, v] = toomre(tmax, level, mass, nstar, rmin, rmax)
  ncore = length(mass); nt = 2^level + 1;
  t = linspace(0, tmax, nt); N = ncore + nstar;
  r = zeros(N,3,nt); v = zeros(N,3,nt); a = zeros(N,3,nt);
  deltat = t(2) - t(1); core{1}=1; core{2}=2;

  % Initialize cores
  r(core{1},1,1)=-2; v(core{1},1,1)=1;
  r(core{2},1,1)=2; r(core{2},2,1)=2; v(core{2},1,1)=-1;

  % Stars around each core
  for i=3:N
    if i<=2+floor(nstar/2)
      theta=2*pi*rand; radius=rmin+(rmax-rmin)*rand;
      r(i,1,1)=r(core{1},1,1)+radius*cos(theta);
      r(i,2,1)=r(core{1},2,1)+radius*sin(theta);
      vc=sqrt(mass(core{1})/radius);
      v(i,1,1)=v(core{1},1,1)-vc*sin(theta);
      v(i,2,1)=v(core{1},2,1)+vc*cos(theta);
    else
      theta=2*pi*rand; radius=rmin+(rmax-rmin)*rand;
      r(i,1,1)=r(core{2},1,1)+radius*cos(theta);
      r(i,2,1)=r(core{2},2,1)+radius*sin(theta);
      vc=sqrt(mass(core{2})/radius);
      v(i,1,1)=v(core{2},1,1)-vc*sin(theta);
      v(i,2,1)=v(core{2},2,1)+vc*cos(theta);
    end
  end
  % ... (rest of the code)
end
